\flyersec{emerald}{clipboard-check}{Bonnes Pratiques}

\begin{enumerate}[label=\textcolor{emerald}{\scriptsize\faCheckCircle},font=\scriptsize,itemsep=1pt]
  \item \textbf{Commencer par Mon \'Ecole}~: la localisation d\'ebloque tout. Ne jamais tenter de contourner.
  \item \textbf{1 campus suffit} si l'\'etablissement n'a qu'un site. Ne pas multiplier sans raison.
  \item \textbf{V\'erifier les cycles avant les classes}~: un cycle activ\'e sans classes = configuration vide.
  \item \textbf{Importer le personnel par Excel} pour gagner du temps. Toujours v\'erifier avant import.
  \item \textbf{Moindre privil\`ege}~: n'attribuer que les modules n\'ecessaires par utilisateur.
  \item \textbf{Calendrier synchronis\'e} par d\'efaut~: ne basculer en perso. que si n\'ecessaire.
  \item \textbf{Sauvegarder explicitement}~: les onglets de config n\'ecessitent un clic \btn{emerald}{Sauvegarder}.
  \item \textbf{P\'eriodes}~: ouvrir en d\'ebut de trimestre, fermer apr\`es saisie compl\`ete des notes.
\end{enumerate}

\textbf{Classes manquantes} $\to$ inscriptions/notes impossibles \quad$\bullet$\quad

\textbf{Personnel absent} $\to$ pas d'attribution cours/horaires \quad$\bullet$\quad
